\newcommand\DrawGrid{
    \begin{tikzpicture}[remember picture, overlay]
      \def\gridwidth{4.5cm}
      \def\gridheight{26cm}
      \def\cellsize{0.75cm}
      \draw[brandLight] 
        ([xshift=-\gridwidth]current page.north east) 
        grid [step=\cellsize] (current page.south east);
    \end{tikzpicture}
}
\usepackage{eso-pic}
\AddToShipoutPictureBG{%
  \ifnotesgrid
    \AtPageLowerLeft{\DrawGrid}%
  \fi
}

\tcbset{
  candlebox/.style={
    enhanced, breakable, sharp corners,
    colback=CandleGray,
    colframe=CandleInk!12,
    borderline west={2.2pt}{0pt}{CandleInk!22},
    boxrule=0.2pt,
    left=6mm, right=6mm, top=3.5mm, bottom=3.5mm,
    before skip=8pt, after skip=10pt

    fonttitle=\bfseries,
    colbacktitle=white,
    coltitle=CandleInk,
    attach boxed title to top left={yshift=-2mm, xshift=0mm},
    boxed title style={sharp corners, colframe=white, boxrule=0pt}
  }
}
